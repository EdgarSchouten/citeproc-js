  <bibliography>
    <option name="subsequent-author-substitute" value=""/>
    <layout>
      <group class="csl-subheading">
        <names variable="author">
          <name and="text"/>
        </names>
      </group>
      <group class="csl-label">
        <date variable="issued">
          <date-part name="year"/>
        </date>
        <text variable="year-suffix"/>
      </group>
      <group class="csl-item">        
        <text variable="title"/>
      </group>
    </layout>
  </bibliography>
</style>
